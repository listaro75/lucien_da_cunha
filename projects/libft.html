int	ft_atoi(const char *nptr)
{
	int	i;
	int	sign;
	int	result;

	i = 0;
	sign = 1;
	result = 0;
	while (nptr[i] == ' ' || (nptr[i] >= 9 && nptr[i] <= 13))
	{
		i++;
	}
	if (nptr[i] == '-')
	{
		sign *= -1;
		i++;
	}
	else if (nptr[i] == '+')
		i++;
	while (nptr[i] >= '0' && nptr[i] <= '9')
	{
		result *= 10;
		result += nptr[i] - '0';
		i++;
	}
	return (result * sign);
}

int	ft_count_world(char const*s, char c)
{
	int	i;
	int	size;

	size = 0;
	i = 0;
	while (s[i] != '\0')
	{
		if (s[i] != c)
		{
			size++;
			while (s[i] != c && s[i] != '\0')
				i++;
		}
		else
			i++;
	}
	return (size);
}

{

char	**ft_split(char const *s, char c)
{
	char	**array;
	int		count;
	int		i;

	i = 0;
	if (!s)
		return (NULL);
	s = skipchar(s, c);
	count = ft_count_world(s, c);
	array = (char **)malloc(sizeof(char *) * (count + 1));
	if (!array)
		return (NULL);
	while (i < count)
	{
		array[i] = ft_copy_word(s, c);
		if (!array[i])
			return (NULL);
		s = skipchar(s + skipword(s, c), c);
		i++;
	}
	array[i] = NULL;
	return (array);
}

{
	size_t	i;
	int		j;
	size_t	temp;

	i = 0;
	if (!needle[i])
		return ((char *)haystack);
	while (haystack[i] && i < len)
	{
		j = 0;
		if (haystack[i] == needle[j])
		{
			temp = i;
			while (haystack[temp] && haystack[temp] == needle[j] && temp < len)
			{
				temp++;
				j++;
				if (j == ft_strlen(needle))

	}

		return (NULL);

		return (ft_strdup(""));

CFLAGS = -Wall -Wextra -Werror

all: $(NAME)

$(NAME): $(OBJECTS)
	$(AR) -r $@ $?

bonus: $(BOBJECTS)
	$(AR) -r $(NAME) $?

%.o: %.c
	$(CC) -c $(CFLAGS) $?

clean:
	rm -f $(OBJECTS) $(BOBJECTS)

fclean: clean
	rm -f $(NAME)

re: fclean all

.PHONY: all bonus clean fclean re
